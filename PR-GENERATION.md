### Example Usage

```typescript
// User detects a break:
const impact = await checkImpact({
  table: 'stg_orders',
  changeType: 'rename',
  newName: 'orders'
});

// AI agent creates fixes:
const fixes = impact.affectedFiles.map(file => ({
  filePath: file.filePath,
  original: `{{ ref('stg_orders') }}`,
  replacement: `{{ ref('orders') }}`
}));

// Generate PR automatically:
const pr = await generatePR({
  title: `fix: Update stg_orders → orders references`,
  description: `
    ## Lineage-MCP Auto-Generated PR
    
    **Reason**: Table rename detected
    - Table: stg_orders → orders
    - Affected files: ${fixes.length}
    - Confidence: high
    
    **Changes**:
    ${fixes.map(f => `- [ ] ${f.filePath}`).join('\n')}
    
    **Generated by**: Lineage-MCP v0.2.0
    **Lineage Report**: [View Impact Analysis](...)
  `,
  changes: fixes,
  labels: ['auto-generated', 'lineage-mcp'],
  assignees: ['data-team']
}, {
  githubToken: process.env.GITHUB_TOKEN,
  owner: 'JCHETAN26',
  repo: 'Lineage-MCP',
  branch: 'main'
});

console.log(`PR created: ${pr.prUrl}`);
```

```typescript
// src/janitor/github-pr.ts (NEW)

import { Octokit } from "octokit";

async function createPullRequest(
  payload: PRPayload,
  config: PRConfig
): Promise<{ url: string; number: number }> {
  const octokit = new Octokit({ auth: config.githubToken });
  
  const response = await octokit.rest.pulls.create({
    owner: config.owner,
    repo: config.repo,
    title: payload.title,
    body: payload.description,
    head: `fix/lineage-${Date.now()}`,
    base: config.branch,
    labels: payload.labels,
    assignees: payload.assignees
  });
  
  return {
    url: response.data.html_url,
    number: response.data.number
  };
}
```

## Current Blocker

PR generation **requires**:
1. Write access to user's GitHub repo
2. Valid GitHub PAT (Personal Access Token)
3. Local git installation

This is an **opt-in** feature (users provide token if they want auto-PR).

## Alternative: Diff-Only Output

Until PR generation is implemented, users can:

```bash
# Get a unified diff
npm run cli -- generate-diff stg_orders --change rename

# Output:
# --- models/marts/order_items.sql
# +++ models/marts/order_items.sql
# -select * from {{ ref('stg_orders') }}
```
